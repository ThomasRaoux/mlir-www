33 static LogicalResult verifyRawBufferOp(T &op) {
34  MemRefType bufferType = op.memref().getType().template cast<MemRefType>();
35  if (bufferType.getMemorySpaceAsInt() != 0)
36  return op.emitOpError(
37  "Buffer ops must operate on a memref in global memory");
38  if (!bufferType.hasRank())
39  return op.emitOpError(
40  "Cannot meaningfully buffer_store to an unranked memref");
41  if (static_cast<int64_t>(op.indices().size()) != bufferType.getRank())
42  return op.emitOpError("Expected " + Twine(bufferType.getRank()) +
43  " indices to memref");
44  return success();
45 }
